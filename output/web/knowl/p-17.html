Suppose we wanted to run the following small program and store the results for later use.
output <- NULL
for(i in 1:5){
  if(i%%2==1){
       output <- c(output, f(i)) ## odd i
  }else{
       output <- c(output, 2*f(i)) ## even i
  }
}
output
plot(output)
plot(output, type='l')
